            turtle.pencolor("brown")
        else:
            turtle.pencolor("green")
        turtle.forward(d)
        turtle.left(r)
        tree(0.7*d,r,s*0.6)
        turtle.right(2*r)
        tree(0.7*d,r,s*0.6)
        turtle.left(r)
        turtle.backward(d)
        turtle.update()
