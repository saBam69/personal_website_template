<p>import turtle<br>
    def tree (d, r , s):<br>
        turtle.pensize(s)<br>
        if d<3 :<br>
            return<br>
        if d>=5:<br>
            turtle.pencolor("brown")<br>
        else:<br>
            turtle.pencolor("green")<br>
        turtle.forward(d)<br>
        turtle.left(r)<br>
        tree(0.7*d,r,s*0.6)<br>
        turtle.right(2*r)<br>
        tree(0.7*d,r,s*0.6)<br>
        turtle.left(r)<br>
        turtle.backward(d)<br>
        turtle.update()<br>
    #turtle.speed(0)<br>
    turtle.tracer(0)<br>
    turtle.left(90)<br>
    tree(70,30,4)<br>
    turtle.mainloop()</p>
