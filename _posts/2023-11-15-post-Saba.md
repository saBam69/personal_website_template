<pre>import turtle
    def tree (d, r , s):
        turtle.pensize(s)
        if d<3 :
            return
        if d>=5:
            turtle.pencolor("brown")
        else:
            turtle.pencolor("green")
        turtle.forward(d)
        turtle.left(r)
        tree(0.7*d,r,s*0.6)
        turtle.right(2*r)
        tree(0.7*d,r,s*0.6)
        turtle.left(r)
        turtle.backward(d)
        turtle.update()
    #turtle.speed(0)
    turtle.tracer(0)
    turtle.left(90)
    tree(70,30,4)
    turtle.mainloop()</pre>
